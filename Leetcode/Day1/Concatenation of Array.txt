class Solution {
public:
    vector<int> getConcatenation(vector<int>& a) {
        vector<int>b;
        int n=a.size();
        for(int i=0;i<a.size();i++){
            b.push_back(a[i]);
        }
        for(int i=0;i<n;i++){
            b.push_back(a[i]);
        }
        return b;
        
    }
};
