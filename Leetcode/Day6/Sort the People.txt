class Solution {
public:
    vector<string> sortPeople(vector<string>& a, vector<int>& h) {
        int n=a.size();
        vector<pair<int,string>> p;
        for( int i=0;i<n;i++){
            p.push_back({h[i],a[i]});
        }
        sort(p.rbegin(),p.rend());
        vector<string>res;
        for(auto it:p){
            res.push_back(it.second);
        }
        return res;
    }
};
