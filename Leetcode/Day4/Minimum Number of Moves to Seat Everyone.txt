class Solution {
public:
    int minMovesToSeat(vector<int>& s, vector<int>& st) {
        int sum=0;
        for(int i=0;i<s.size();i++){
            sort(s.begin(),s.end());
            sort(st.begin(),st.end());
        }
        for(int i=0;i<s.size();i++){
             sum=sum+abs(st[i]-s[i]);
        }
        return sum;
    }
};
